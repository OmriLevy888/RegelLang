Literal

Syntax

Literal := TextLiteral | NumeralLiteral | BooleanLiteral

TextLiteral := RegexLiteral | StringLiteral | CharLiteral
RegexLiteral := '$' StringLiteral
StringLiteral := StringDelim CharValue* StringDelim
StringDelim := " CharLiteral := CharDelim CharValue CharDelim
CharDelim := ' CharValue := Alpha | DecimalDigit | SpecialDigit | EscapeCharacter

NumeralLiteral := RealLiteral | NaturalLiteral
RealLiteral := [RealNumber | DecimalNumber] ['d' | 'f']
RealNumber := DecimalDigit* '.' DecimalDigit*

NaturlaLiteral := IntLiteral | UintLiteral
IntLiteral := NaturalNumber IntPostfix?
UintLiteral := NaturalNumber UintPostfix
IntPostfix := 'i' | 'i8' | 'i16' | 'i32' | 'i64'
UintPostfix := 'u' | 'u8' | 'u16' | 'u32' | 'u64'
DecimalNumber := DecimalDigit+
BinaryNumber := BinaryPrefix BinaryDigit+
BinaryPrefix := '0b' | '0B'
OctalNumber := OctalPrefix OctalDigit+
OctalPrefix := '0o' | '0O'
HexNumber := HexPrefix HexDigit+
HexPrefix := '0x' | '0X'

BooleanLiteral := 'true' | 'false'

Description

Literals are constant values. 'i' and 'u' suffixes are aliases to 'i32' and 'u32'. More information about each type can be found at the corresponding page.
